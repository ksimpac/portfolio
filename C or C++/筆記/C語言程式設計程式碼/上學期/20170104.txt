void bubblesort(int data[], int size)
{
	for (int i = 0; i <= size; i++)
	{
		for (int j = i + 1; j <= size - 1; j++)
		{
			if (data[i] > data[j])
			{
				int temp = data[i];
				data[i] = data[j];
				data[j] = temp;
			}
		}
	}
}
int times(int data[], int size)
{
	int a[101] = { 0 };

	for (int i = 0; i <= size; i++)
	{
		a[data[i]]++;
	}

	int max_score = 0, max_times = 0;

	for (int j = 0; j <= 101; j++)
	{
		if (a[j] > max_times)
		{
			max_score = j;
			max_times = a[j];
		}
	}

	return max_score;
}
