#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

int main(void)
{
	char string[] = "d1k4FuqQuOcC12W4C5";
	int boolean = 0,lower = 0,upper = 0,number = 0;
    
	/* 
	   string為字串
	   boolean 判斷是與否
	   lower,upper,number 分別計算小寫,大寫及數字出現的次數
	*/

	for (int i = 0; i <= 17; i++) //從第一個字到最後一個字
	{
		boolean = islower(string[i]); //判斷是否為小寫
		
		if (boolean != 0) //boolean是否非0值
		{
			lower += 1; //小寫出現次數+1
			string[i] = toupper(string[i]); //小寫轉成大寫
			continue; //跑下一個字
		}

		boolean = isupper(string[i]);

		if (boolean != 0) //boolean是否非0值
		{
			upper += 1; //小寫出現次數+1
			string[i] = tolower(string[i]); //大寫轉成小寫
			continue; //跑下一個字
		}

		boolean = isdigit(string[i]);

		if (boolean != 0) //boolean是否非0值
		{
			number += 1; //數字出現次數+1
			continue; //跑下一個字
		}
		
	}

	printf("字串中小寫%d個 大寫%d個 數字%d個\n", lower, upper, number);
	system("PAUSE");
	return 0;
}
