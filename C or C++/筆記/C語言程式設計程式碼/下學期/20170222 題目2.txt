#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(void)
{
	char s1[] = "abcdefabacdefgh";
	char s2[] = "def";
	int times = 0;
	char *ptr = s1;

        /*
          times計算s2在s1出現的次數
          ptr指向s1的第一個字元
        */

	for (int i = 0; i <= 14;i++)
	{
		if (strncmp(ptr, s2, 3) == 0) //ptr前三個字元是否為s2
		{
			times += 1; //出現s2就+1
		}
		ptr++; //ptr指向s1的下一個字元
	}

	printf("def總共出現%d次\n", times);
	system("PAUSE");
	return 0;
}
