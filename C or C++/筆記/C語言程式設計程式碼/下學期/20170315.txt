void convert_to_binary(int number) //轉成二進位函數(副程式)
{

        /*
           有需要知道原理的請點開下面網站
           https://drive.google.com/file/d/0ByAvpptIcHnISFphNFM0MldYZUU/view?usp=sharing
        */

	int mask = 1 << 31;

	for (int i = 1; i <= 32; i++)
	{
		if (number & mask != 0)
		{
			printf("%c", '1');
		}
		else
		{
			printf("%c", '0');
		}

		number <<= 1; //number往左移一位元

		if (i % 8 == 0) 
		{
			printf(" "); //做格式而已
		}
	}

	printf("\n"); //做格式而已
}
